// Parameters
#pragma parameter SCAN_STR "Scanline Intensity" 0.35 0.0 1.0 0.05
#pragma parameter SCAN_ZOOM "Scanline Zoom" 2.0 1.0 10.0 0.5

// Code
#ifdef PARAMETER_UNIFORM
    uniform float SCAN_STR, SCAN_ZOOM;


// 1. حساب الموجة (جيب الزاوية بيعمل خطوط ناعمة وسريعة جداً)
// نستخدم gl_FragCoord عشان الخطوط تبقى ثابتة على الشاشة مهما حصل
float scan = sin(gl_FragCoord.y * (6.283 / SCAN_ZOOM)) * 0.5 + 0.5;

// 2. دمج مباشر في الصورة
res *= mix(1.0, scan, SCAN_STR);
